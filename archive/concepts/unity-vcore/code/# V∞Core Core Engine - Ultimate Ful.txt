import math
import numpy as np
import random
from typing import List, Tuple
import pennylane as qml  # Optional quantum (comment out if no install)

class VCoreUltimate:
    def __init__(self, n: int = 1000000, chi: float = 2.0, dimension: int = 11,
                 gamma: float = 3.0, quantum: bool = False):
        self.n = n
        self.chi = chi
        self.dimension = max(3, min(11, dimension))
        self.gamma = gamma
        self.beta = 0.35
        self.e_threshold = 1e-6
        
        # Vectorized state (performance)
        self.radii = np.logspace(-1, 1, n)  # non-symmetric flow
        self.k = np.full(n, 12, dtype=np.int32)
        
        # Quantum optional
        self.quantum = quantum
        if quantum:
            self.qubits_max = 32
            self.dev = qml.device("default.qubit", wires=self.qubits_max)
            self.pool = self._build_pauli_pool()
        
        # Classical optimizers params
        self.sa_temp = 10.0
        self.ga_pop = 50
        self.pso_particles = 30

    # ==================== Core Mechanics ====================
    def avg_k(self) -> float:
        return self.k.mean()

    def compute_e(self) -> float:
        avg_k = self.avg_k() or 1
        d_factor = self.dimension / 11.0
        exp = 1.6 - 0.4 * d_factor  # high D weaker diminishing
        savings = self.beta / (avg_k ** exp)
        alpha_eff = 1.9 * (1 + self.chi * d_factor)
        return alpha_eff - savings

    def select_k(self) -> int:
        base = 2 if self.chi < 0.83 else 3
        return base + int(self.dimension > 8) + int(self.dimension > 10)

    # ==================== Angles & Flow ====================
    def symmetric_angles(self, k: int) -> Tuple[float, float]:
        if k < 2: return 0.0, 0.0
        theta = math.acos(-1.0 / k)
        phi = math.acos(1.0 / k)
        return math.degrees(theta), math.degrees(phi)

    def non_symmetric_angles(self, parent_r: float, daughter_rs: List[float]) -> List[float]:
        if abs(parent_r**self.gamma - sum(r**self.gamma for r in daughter_rs)) > 1e-6:
            raise ValueError("Flow not conserved")
        angles = []
        for r_d in daughter_rs:
            ratio = r_d / parent_r
            base = 90.0 if self.chi > 1.0 or self.dimension > 9 else 120.0
            theta = base * (1 - ratio) + 90.0 * ratio
            theta *= (11.0 / self.dimension)
            angles.append(theta)
        return angles

    def overlap_factor(self, k: int) -> float:
        return 1.0 - 1.0 / k if k > 1 else 1.0

    # ==================== Classical Optimizers ====================
    def simulated_annealing_tunnel(self, cluster: np.ndarray):
        temp = self.sa_temp
        current = self.k[cluster].copy()
        best = current.copy()
        current_cost = self._flare_cost(current)
        best_cost = current_cost
        
        while temp > 1e-4:
            candidate = current + np.random.randint(-2, 3, size=len(cluster))
            candidate = np.clip(candidate, 10, 300)
            cost = self._flare_cost(candidate)
            if cost < current_cost or random.random() < math.exp((current_cost - cost)/temp):
                current = candidate
                current_cost = cost
                if cost < best_cost:
                    best = candidate
                    best_cost = cost
            temp *= 0.95
        self.k[cluster] = best

    def genetic_algorithm_tunnel(self, cluster: np.ndarray):
        pop = [self.k[cluster].copy() for _ in range(self.ga_pop)]
        for ind in pop[1:]:
            ind += np.random.randint(-2, 3, size=len(cluster))
            ind = np.clip(ind, 10, 300)
        
        for _ in range(100):
            costs = [self._flare_cost(ind) for ind in pop]
            elite = np.argsort(costs)[:5]
            new_pop = [pop[i].copy() for i in elite]
            while len(new_pop) < self.ga_pop:
                p1, p2 = random.choices(pop, weights=[1/(c+1) for c in costs], k=2)
                child = p1.copy()
                cp = random.randint(0, len(cluster))
                child[cp:] = p2[cp:]
                child += np.random.randint(-1, 2, size=len(cluster)) * (np.random.rand(len(cluster)) < 0.1)
                child = np.clip(child, 10, 300)
                new_pop.append(child)
            pop = new_pop
        
        best = pop[np.argmin([self._flare_cost(ind) for ind in pop])]
        self.k[cluster] = best

    def particle_swarm_tunnel(self, cluster: np.ndarray):
        particles = np.tile(self.k[cluster], (self.pso_particles, 1))
        velocities = np.random.randint(-2, 3, size=particles.shape)
        pbest = particles.copy()
        pbest_cost = np.array([self._flare_cost(p) for p in pbest])
        gbest = pbest[np.argmin(pbest_cost)].copy()
        
        for _ in range(100):
            r1, r2 = np.random.rand(2, self.pso_particles, len(cluster))
            velocities = (0.7 * velocities +
                          1.5 * r1 * (pbest - particles) +
                          1.5 * r2 * (gbest - particles))
            particles += velocities.astype(int)
            particles = np.clip(particles, 10, 300)
            costs = np.array([self._flare_cost(p) for p in particles])
            improved = costs < pbest_cost
            pbest[improved] = particles[improved]
            pbest_cost[improved] = costs[improved]
            if min(costs) < min(pbest_cost):
                gbest = particles[np.argmin(costs)].copy()
        
        self.k[cluster] = gbest

    def _flare_cost(self, k_array: np.ndarray) -> float:
        avg_k = k_array.mean() or 1
        d_factor = self.dimension / 11.0
        return 4 * math.pi / (avg_k ** (1.2 - 0.3 * d_factor)) * (1 + 0.5 / self.chi)

    # ==================== Quantum Optimizers (Optional) ====================
    if quantum:
        def _build_pauli_pool(self):
            pool = []
            for i in range(self.qubits_max):
                pool.append(qml.PauliX(i))
                for j in range(i+1, self.qubits_max):
                    pool.append(qml.PauliY(i) @ qml.PauliY(j))
            return pool
        
        def adapt_vqe_tunnel(self, cluster_indices: np.ndarray):
            n_q = len(cluster_indices)
            if n_q > self.qubits_max: return
            
            # Ising proxy H_flare
            coeffs, paulis = [], []
            for i in range(n_q):
                for j in range(i+1, n_q):
                    coeffs.append(1.0)
                    paulis.append(qml.PauliZ(i) @ qml.PauliZ(j))
            H = qml.Hamiltonian(coeffs, paulis)
            
            ops = []
            params = np.array([], requires_grad=True)
            
            @qml.qnode(self.dev)
            def energy(params, ops):
                for i in range(n_q): qml.Hadamard(wires=i)
                for theta, op in zip(params, ops):
                    qml.exp(-1j * theta * op)
                return qml.expval(H)
            
            while True:
                grads = []
                for op in self.pool:
                    if op not in ops:
                        comm = qml.operation.commutator(H, op)
                        grad = abs(qml.expval(comm)(params, ops))
                        grads.append((grad, op))
                if not grads or max(g[0] for g in grads) < 1e-4: break
                
                best_op = max(grads, key=lambda x: x[0])[1]
                ops.append(best_op)
                params = np.append(params, 0.0)
                
                opt = qml.AdamOptimizer()
                for _ in range(50):
                    params = opt.step(lambda p: energy(p, ops), params)
            
            optimal_k = int(-energy(params, ops) * 10 + 100)
            self.k[cluster_indices] = np.clip(optimal_k, 10, 300)

    # ==================== Main Ripple ====================
    def ripple_step(self):
        e = self.compute_e()
        if abs(e) < self.e_threshold: return False
        
        # Hybrid tunnel trigger (stuck proxy)
        if np.random.rand() < 0.15:
            cluster = np.random.choice(self.n, size=min(1000, self.n//20), replace=False)
            # Classical hybrid
            choice = random.choice(['sa', 'ga', 'pso'])
            if choice == 'sa':
                self.simulated_annealing_tunnel(cluster)
            elif choice == 'ga':
                self.genetic_algorithm_tunnel(cluster)
            else:
                self.particle_swarm_tunnel(cluster)
            if self.quantum:
                self.adapt_vqe_tunnel(cluster)
        
        k_new = self.select_k()
        core = slice(0, self.n//3)
        side = slice(self.n//3, 2*self.n//3)
        remote = slice(2*self.n//3, self.n)
        
        if e > 0:
            self.k[core] += k_new
            self.k[side] += 2
        else:
            prune_amt = int(3 * (1 - self.chi))
            self.k[remote] = np.maximum(self.k[remote] - prune_amt, 10)
        
        self.beta *= 0.98
        return True

    def run_cascade(self) -> Tuple[float, int]:
        depth = 0
        while self.ripple_step():
            depth += 1
            if depth > 300: break
        return self.avg_k(), depth
